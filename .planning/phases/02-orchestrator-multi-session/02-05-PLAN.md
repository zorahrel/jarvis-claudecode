    import type { SnapshotEntry, Confidence, AutoPilotDecision, AutoPilotSkipReason } from "./types.js";
    import type { AutoPilotConfig } from "../../types/config.js";

    const DEFAULT_CONFIG: AutoPilotConfig = {
      enabled: false,
      daily_token_cap: 100_000,
      confidence_floor: "high",
      bypass_lock_conflict: false,
      log_only: false,
    };

    export function loadAutopilotConfig(rootConfig: { auto_pilot?: Partial<AutoPilotConfig> }): AutoPilotConfig {
      const ap = rootConfig.auto_pilot ?? {};
      return { ...DEFAULT_CONFIG, ...ap };
    }

    export interface BudgetState {
      tokensUsedToday: number;
    }

    export interface CheckBudgetOptions {
      fetchFn?: typeof fetch;
      baseUrl?: string;
    }

    export async function checkBudget(opts: CheckBudgetOptions = {}): Promise<BudgetState> {
      const f = opts.fetchFn ?? fetch;
      const url = `${opts.baseUrl ?? "http://localhost:3340"}/api/sessions/aggregate`;
      try {
        const r = await f(url);
        if (!r.ok) return { tokensUsedToday: 0 };
        const j = await r.json() as { tokens_today?: number };
        return { tokensUsedToday: j.tokens_today ?? 0 };
      } catch {
        return { tokensUsedToday: 0 };
      }
    }

    function meetsConfidenceFloor(c: Confidence, floor: "high" | "medium"): boolean {
      if (floor === "high") return c === "high";
      if (floor === "medium") return c === "high" || c === "medium";
      return false;
    }

    export async function shouldAutoInject(
      entry: SnapshotEntry,
      cfg: AutoPilotConfig,
      budget: BudgetState,
    ): Promise<AutoPilotDecision> {
      const ts = Date.now();
      // Spawn guard — never auto-inject on router-spawned sessions (avoid recursion)
      if (process.env.JARVIS_SPAWN === "1") {
        // (W6 FIX) Removed `as never` cast — TypeScript infers the literal "spawn_guard"
        // correctly because AutoPilotSkipReason is a union of literals.
        return { pid: entry.pid, applied: false, reason: "spawn_guard", ts };
      }
      if (!cfg.enabled) {
        return { pid: entry.pid, applied: false, reason: "disabled", ts };
      }
      if (entry.status !== "awaiting_user_input") {
        return { pid: entry.pid, applied: false, reason: "wrong_status", ts };
      }
      if (entry.action.type !== "inject" || !("text" in entry.action) || !entry.action.text) {
        return { pid: entry.pid, applied: false, reason: "wrong_status", ts };
      }
      if (!meetsConfidenceFloor(entry.confidence as Confidence, cfg.confidence_floor)) {
        return { pid: entry.pid, applied: false, reason: "low_confidence", ts };
      }
      if (!entry.tmux) {
        return { pid: entry.pid, applied: false, reason: "no_tmux", ts };
      }
      if (entry.conflict !== null && !cfg.bypass_lock_conflict) {
        return { pid: entry.pid, applied: false, reason: "lock_conflict", ts };
      }
      if (budget.tokensUsedToday >= cfg.daily_token_cap) {
        return { pid: entry.pid, applied: false, reason: "budget_exceeded", ts };
      }
      if (cfg.log_only) {
        return { pid: entry.pid, applied: false, reason: "log_only_mode", action: entry.action, ts };
      }
      // Eligible
      return { pid: entry.pid, applied: true, reason: "applied", action: entry.action, ts };
    }

    export interface AutopilotHandle {
      active: boolean;
      stop: () => void;
    }

    let activeHandle: AutopilotHandle | null = null;

    export function registerAutopilotHook(cfg: AutoPilotConfig): AutopilotHandle {
      // Do NOT install the OS-level UserPromptSubmit hook here — that lives in
      // .claude/hooks/UserPromptSubmit-orchestrator-autopilot.sh and is enabled
      // by user action (manual config). This function only manages the in-process
      // listener that the hook script will reach via /api/orchestrator/autopilot/tick.
      if (!cfg.enabled) {
        activeHandle = { active: false, stop: () => undefined };
        return activeHandle;
      }
      activeHandle = { active: true, stop: () => { activeHandle = { active: false, stop: () => undefined }; } };
      return activeHandle;
    }

    export function getAutopilotHandle(): AutopilotHandle | null {
      return activeHandle;
    }
