POST /api/sessions/:pid/inject contract:
- Body: `{ text: string, source: "user-approved" | "auto" | "skill", confidence?: "low"|"medium"|"high", reason?: string, force?: boolean }`
- 200 → `{ ok: true, paneId: "%2", auditTs: 1715000000 }`
- 409 (no tmux) → `{ error: "no_tmux", message: "session not under tmux — bare TTY is read-only" }`
- 409 (lock conflict + force=false) → `{ error: "lock_conflict", message: "cwd shared with pid <X>", conflictPid: <X> }`
- 404 (pane lost mid-flight) → `{ error: "pane_lost", message: "pane %2 disappeared between resolve and send" }`

    1. **Create `router/src/services/orchestrator/tmuxMap.ts`**:

    ```typescript
    import { execFile as execFileCb } from "child_process";
    import { promisify } from "util";

    const execFileDefault = promisify(execFileCb);

    type ExecFn = (cmd: string, args: string[]) => Promise<{ stdout: string; stderr: string }>;

    export interface PaneRow {
      pid: number;
      session: string;
      pane: string;
      windowIndex: number;
      active: boolean;
    }

    export async function listAllPanes(execFn: ExecFn = execFileDefault): Promise<PaneRow[]> {
      let stdout: string;
      try {
        const r = await execFn("tmux", ["list-panes", "-aF", "#{pane_pid} #{session_name} #{pane_id} #{window_index} #{pane_active}"]);
        stdout = r.stdout;
      } catch {
        return [];
      }
      return stdout.trim().split("\n").filter((l) => l.length > 0).map((line) => {
        const parts = line.split(" ");
        return {
          pid: parseInt(parts[0] ?? "0", 10),
          session: parts[1] ?? "",
          pane: parts[2] ?? "",
          windowIndex: parseInt(parts[3] ?? "0", 10),
          active: parts[4] === "1",
        };
      });
    }

    export async function findPaneForPid(targetPid: number, execFn: ExecFn = execFileDefault): Promise<{ session: string; pane: string } | null> {
      const panes = await listAllPanes(execFn);
      if (panes.length === 0) return null;
      let cur = targetPid;
      for (let i = 0; i < 50 && cur > 1; i++) {
        const hit = panes.find((p) => p.pid === cur);
        if (hit) return { session: hit.session, pane: hit.pane };
        try {
          const { stdout } = await execFn("ps", ["-o", "ppid=", "-p", String(cur)]);
          const ppid = parseInt(stdout.trim(), 10);
          if (!ppid || ppid === cur) break;
          cur = ppid;
        } catch { return null; }
      }
      return null;
    }

    export async function sendKeys(paneId: string, text: string, execFn: ExecFn = execFileDefault): Promise<void> {
      const lines = text.split("\n");
      const args: string[] = ["send-keys", "-t", paneId, "--"];
      for (const line of lines) {
        args.push(line);
        args.push("Enter");
      }
      await execFn("tmux", args);
    ```
